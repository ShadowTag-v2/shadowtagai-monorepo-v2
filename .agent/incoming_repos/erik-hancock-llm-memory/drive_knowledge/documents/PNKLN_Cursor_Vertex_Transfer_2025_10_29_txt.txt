# Minimal, testable reference for pixel-layer stego that targets mid-band DCT.
# NOTE: This is a compact reference; production version belongs in shadowtag_v2/.

import cv2, numpy as np, hashlib, json, os, tempfile

def prompt_hash(prompt: str) -> bytes:
    return hashlib.sha256(prompt.encode("utf-8")).digest()[:16]  # 128-bit

def _embed_block(block, bit):
    # DCT -> tweak a mid-band coefficient parity -> inverse DCT
    dct = cv2.dct(block.astype(np.float32))
    i, j = 3, 2  # chosen mid-band indices; tune during robustness trials
    coeff = dct[i, j]
    parity = int(abs(coeff)) & 1
    if parity != bit:
        dct[i, j] += np.sign(coeff) or 1.0
    return cv2.idct(dct).clip(0, 255).astype(np.uint8)

def embed_frame(frame: np.ndarray, bits: np.ndarray, blk=8):
    h, w = frame.shape[:2]
    yuv = cv2.cvtColor(frame, cv2.COLOR_BGR2YUV)
    Y = yuv[:, :, 0]
    bit_idx = 0
    for y in range(0, h - blk + 1, blk):
        for x in range(0, w - blk + 1, blk):
            if bit_idx >= len(bits): break
            Y[y:y+blk, x:x+blk] = _embed_block(Y[y:y+blk, x:x+blk], int(bits[bit_idx]))
            bit_idx += 1
        if bit_idx >= len(bits): break
    yuv[:, :, 0] = Y
    return cv2.cvtColor(yuv, cv2.COLOR_YUV2BGR)

def bytes_to_bits(b: bytes):
    return np.unpackbits(np.frombuffer(b, dtype=np.uint8))

def embed_video(in_path, out_path, prompt):
    cap = cv2.VideoCapture(in_path)
    fps = cap.get(cv2.CAP_PROP_FPS) or 30.0
    w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    fourcc = cv2.VideoWriter_fourcc(*"mp4v")
    out = cv2.VideoWriter(out_path, fourcc, fps, (w, h))

    bits = bytes_to_bits(prompt_hash(prompt))
    frame_idx = 0
    success, frame = cap.read()
    while success:
        out.write(embed_frame(frame, bits))
        frame_idx += 1
        success, frame = cap.read()
    cap.release(); out.release()
    return {"ok": True, "frames": frame_idx, "w": w, "h": h, "fps": fps}

print("Loaded pixel-layer reference: embed_video(in, out, prompt)")

import numpy as np, soundfile as sf, hashlib
from scipy.signal import butter, lfilter

def spread_spectrum(bits: np.ndarray, sr=48000, dur=1.0, f0=19500.0):
    t = np.linspace(0, dur, int(sr*dur), endpoint=False)
    carrier = np.sin(2*np.pi*f0*t)
    chip = (bits*2-1).astype(float)  # BPSK chips
    reps = int(np.ceil(len(t)/len(chip)))
    chips = np.repeat(chip, reps)[:len(t)]
    return (0.02 * carrier * chips).astype(np.float32)

def butter_hp(sig, sr, fc=18000.0, order=4):
    b, a = butter(order, fc/(sr/2), btype='high')
    return lfilter(b, a, sig)

def embed_ultrasonic(wav_in, wav_out, prompt, sr=48000):
    bits = np.unpackbits(np.frombuffer(hashlib.sha256(prompt.encode()).digest()[:16], dtype=np.uint8))
    carrier = spread_spectrum(bits, sr=sr, dur=1.5, f0=19500.0)
    audio, sr0 = sf.read(wav_in)
    if sr0 != sr:
        raise ValueError("Resample to 48kHz before embedding for robustness.")
    if audio.ndim == 1: audio = np.stack([audio, audio], axis=1)
    pad = np.zeros((max(0, len(carrier)-len(audio)), 2), dtype=np.float32)
    carrier2 = np.stack([carrier, carrier], axis=1)
    mixed = (audio.astype(np.float32) + carrier2[:len(audio)] if len(audio)>=len(carrier2) else
             np.vstack([audio.astype(np.float32), pad]) + np.vstack([carrier2, np.zeros_like(pad)]))
    sf.write(wav_out, mixed, sr)
    return {"ok": True, "sr": sr, "added": "ultrasonic spread-spectrum tag"}

print("Loaded audio-layer reference: embed_ultrasonic(in_wav, out_wav, prompt)")

from web3 import Web3
import hashlib, json, os

def polygon_receipt(rpc_url:str, private_key:str, prompt:str):
    # WARNING: Example only; use GCP Secret Manager in production
    w3 = Web3(Web3.HTTPProvider(rpc_url))
    acct = w3.eth.account.from_key(private_key)
    h = hashlib.sha256(prompt.encode("utf-8")).hexdigest()
    tx = {
        "to": acct.address,
        "value": 0,
        "data": Web3.to_bytes(text=h),
        "nonce": w3.eth.get_transaction_count(acct.address),
        "gas": 50000,
        "maxFeePerGas": w3.to_wei(30, "gwei"),
        "maxPriorityFeePerGas": w3.to_wei(1, "gwei"),
        "chainId": 137,
        "type": 2,
    }
    signed = acct.sign_transaction(tx)
    txh = w3.eth.send_raw_transaction(signed.rawTransaction).hex()
    return {"ok": True, "tx": txh, "hash": h}

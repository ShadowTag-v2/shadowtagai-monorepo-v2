import time, threading
N = 40_000_00
def work():
    s = 0
    for i in range(1, N):
        s += (i % 97)
    return s

def run_threads(t=4):
    outs = [0]*t
    def runner(k): outs[k] = work()
    ts = [threading.Thread(target=runner, args=(k,)) for k in range(t)]
    t0 = time.time()
    [x.start() for x in ts]
    [x.join() for x in ts]
    dt = time.time() - t0
    print(f"threads={t}, time={dt:.2f}s")

scripts/ingest/moondream_ingest.ts

// pnpm add moondream fs-extra fast-glob undici
import fg from "fast-glob";
import fs from "fs-extra";
import { request } from "undici";
import path from "node:path";
// import { Moondream } from "moondream"; // if using the local lib/binary
// const md = new Moondream({ model: "moondream2" });

const ROOTS = (process.env.INGEST_ROOTS || "C:/Users;E:/;F:/").split(";");
const OUT   = process.env.INGEST_OUT || "ingest/out/downloads.jsonl";
const SEEN  = process.env.INGEST_SEEN || "ingest/out/.seen.txt"; // simple seen list; keep SQLite if you prefer
const GPTRAM = process.env.GPTRAM_URL || ""; // e.g. http://localhost:8765

const exts = [".png",".jpg",".jpeg",".pdf",".tiff",".bmp",".webp",".gif",".heic",".txt",".md",".csv",".json",".html"];

const seen = new Set<string>(fs.existsSync(SEEN) ? fs.readFileSync(SEEN,"utf8").split("\n").filter(Boolean) : []);

function sha256(buf: Buffer){ return require("crypto").createHash("sha256").update(buf).digest("hex"); }

async function parseWithMoondream(p: string){
  // Minimal: call a local service or the lib (pick one approach)
  // 1) If you run a tiny local HTTP wrapper for Moondream:
  // const r = await request("http://127.0.0.1:7777/extract", { method:"POST", body: fs.createReadStream(p) });
  // return await r.body.json();

  // 2) Or use an embedded lib/binary (pseudo):
  // const res = await md.extract(p);
  // return res;

  // Fallback: plain text for non-images
  const ext = path.extname(p).toLowerCase();
  if (ext === ".txt" || ext === ".md" || ext === ".csv" || ext === ".json" || ext === ".html"){
    const text = await fs.readFile(p, "utf8");
    return { text, json: null, meta: { mode: "plain" } };
  }
  return { text: "", json: null, meta: { mode: "skipped" } };
}

async function putGPTRAM(key: string, payload: any){
  if (!GPTRAM) return;
  try{
    await request(`${GPTRAM}/put`, {
      method: "POST",
      headers: { "content-type":"application/json" },
      body: JSON.stringify({ key, text: payload.text ?? "", meta: payload.json ?? payload.meta ?? {}, ts: Math.floor(Date.now()/1000) })
    });
  }catch{}
}

async function main(){
  await fs.ensureDir(path.dirname(OUT));
  const out = fs.createWriteStream(OUT, { flags: "a" });

  for (const root of ROOTS){
    const files = await fg(["**/*"], { cwd: root, dot:false, onlyFiles:true, absolute:true, followSymbolicLinks:false });
    for (const p of files){
      const ext = path.extname(p).toLowerCase();
      if (!exts.includes(ext)) continue;

      const buf = await fs.readFile(p).catch(()=>null);
      if (!buf) continue;
      const h = sha256(buf);
      if (seen.has(h)) continue;

      const parsed = await parseWithMoondream(p);
      const rec = {
        sha256: h,
        path: p,
        ext,
        size: buf.length,
        mtime: Math.floor((await fs.stat(p)).mtimeMs/1000),
        text: parsed.text || undefined,
        data: parsed.json || undefined,
        meta: parsed.meta || undefined,
        now: Math.floor(Date.now()/1000),
        source: "moondream-ingest"
      };

      out.write(JSON.stringify(rec)+"\n");
      await putGPTRAM(h, { text: rec.text || "", json: rec.data || rec.meta });

      seen.add(h);
    }
  }
  out.end();
  await fs.outputFile(SEEN, [...seen].join("\n"));
  console.log(`[ingest] wrote → ${OUT}  new=${seen.size}`);
}

main().catch(e=>{ console.error(e); process.exit(1); });

    p.on("exit", (c) => c === 0 ? resolve() : reject(new Error(`exit ${c}`)));

async function run(npmScript) {
  await new Promise((resolve, reject) => {
    const p = spawn(process.platform === "win32" ? "npm.cmd" : "npm", ["run", npmScript], { stdio: "inherit" });
    p.on("exit", (c) => c === 0 ? resolve() : reject(new Error(`${npmScript} failed`)));
  });
}

// pseudo: run in your Node orchestrator, called from a Cursor task
import { chromium } from 'playwright';
import { callComputerUse } from './llm'; // wraps gemini computer-use

type Step = { action: 'click'|'type'|'navigate'|'wait'; selector?: string; text?: string; url?: string };

export async function runComputerUseJob(job: {goal: string, allow: string[]}) {
  const browser = await chromium.launch();
  const page = await browser.newPage();
  const context = { allow: job.allow, history: [] as Step[] };

  // seed page if needed
  await page.goto('https://vercel.com/dashboard');

  for (let i=0; i<60; i++) { // hard cap ~10 min
    const screenshot = await page.screenshot({ fullPage: true });
    const plan = await callComputerUse({
      goal: job.goal,
      allowList: job.allow,
      screenshot, // the model “sees” what’s on screen
      priorSteps: context.history
    });

    for (const step of plan.steps as Step[]) {
      if (step.action === 'navigate' && step.url) await page.goto(step.url);
      if (step.action === 'click' && step.selector) await page.click(step.selector);
      if (step.action === 'type' && step.selector) await page.fill(step.selector, step.text ?? '');
      if (step.action === 'wait') await page.waitForTimeout(1000);
      context.history.push(step);
    }
    if (plan.done) break;
  }

  const report = await page.title();
  await browser.close();
  return { ok: true, report, steps: context.history };
}

computer-use/executor.py

from __future__ import annotations
from playwright.sync_api import Page

VW, VH = 1440, 900   # recommended viewport

def _px(x_norm: float, y_norm: float):
    # API uses ~0..999 normalized coords; clamp to viewport.
    x = int(max(0, min(VW-1, x_norm * VW / 999)))
    y = int(max(0, min(VH-1, y_norm * VH / 999)))
    return x, y

def run_action(page: Page, name: str, args: dict) -> str:
    if name == "open_web_browser":
        # noop (we manage browser externally)
        return "ok"
    if name == "navigate":
        url = args.get("url") or "about:blank"
        page.goto(url, wait_until="domcontentloaded")
        return f"nav:{url}"
    if name == "click_at":
        x, y = _px(args.get("x", 0), args.get("y", 0))
        page.mouse.click(x, y)
        return f"click:{x},{y}"
    if name == "type_text_at":
        x, y = _px(args.get("x", 0), args.get("y", 0))
        page.mouse.click(x, y)
        text = args.get("text", "")
        page.keyboard.type(text, delay=10)
        if args.get("press_enter"):
            page.keyboard.press("Enter")
        return f"type:{len(text)}"
    if name == "wait":
        ms = int(args.get("ms", 500))
        page.wait_for_timeout(ms)
        return f"wait:{ms}"
    if name == "back":
        page.go_back()
        return "back"
    if name == "forward":
        page.go_forward()
        return "forward"
    # Extend with other predefined functions as needed.
    return f"unknown:{name}"

computer-use/agent.py

from __future__ import annotations
import os, sys, base64, json, time
from pathlib import Path
from typing import List
from google import genai
from google.genai import types
from playwright.sync_api import sync_playwright
from .executor import run_action, VW, VH
from .audit import log
import yaml

MODEL = os.environ.get("CU_MODEL", "gemini-2.5-computer-use-preview-10-2025")
GOAL  = os.environ.get("CU_GOAL", "Go to example.com and capture the H1 text.")
START = os.environ.get("CU_START_URL", "https://example.com")
ALLOW = yaml.safe_load(Path(__file__).with_name("allowlist.yaml").read_text(encoding="utf-8"))
MAX_TURNS = int(ALLOW.get("limits", {}).get("max_turns", 10))
DOMAINS   = set(ALLOW.get("domains", []))
BLOCK_URL_SUBSTR = set(ALLOW.get("block", {}).get("url_contains", []))

def _allowed(url: str) -> bool:
    from urllib.parse import urlparse
    host = (urlparse(url).hostname or "").lower()
    if DOMAINS and not any(host.endswith(d) for d in DOMAINS):
        return False
    return not any(s in url for s in BLOCK_URL_SUBSTR)

def _part_image(png_bytes: bytes) -> types.Part:
    return types.Part.from_bytes(data=png_bytes, mime_type="image/png")

def main() -> int:
    api_key = os.environ.get("GOOGLE_API_KEY")
    if not api_key:
        print("GOOGLE_API_KEY not set", file=sys.stderr); return 2

    client = genai.Client(api_key=api_key)

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        ctx = browser.new_context(viewport={"width": VW, "height": VH})
        page = ctx.new_page()
        page.goto(START, wait_until="domcontentloaded")
        log("start", goal=GOAL, start=START, model=MODEL)

        turns = 0
        while turns < MAX_TURNS:
            turns += 1

            png = page.screenshot(full_page=False)
            cfg = types.GenerateContentConfig(
                tools=[types.Tool(computer_use=types.ComputerUse())]
                # You can exclude risky predefined functions if needed:
                # , excluded_predefined_functions=["download_file", ...]
            )

            resp = client.models.generate_content(
                model=MODEL,
                contents=[types.Content(role="user", parts=[
                    types.Part(text=GOAL),
                    _part_image(png),
                ])],
                config=cfg
            )
            cand = resp.candidates[0]
            acts = []
            for part in cand.content.parts:
                if hasattr(part, "function_call") and part.function_call:
                    acts.append(part.function_call)

            # Safety decision may ask for confirmation on risky ops:
            if getattr(cand, "safety_ratings", None):
                log("safety", ratings=[sr.__dict__ for sr in cand.safety_ratings])

            if not acts:
                log("done", reason="no-actions")
                break

            for fc in acts:
                name = fc.name
                args = dict(fc.args or {})
                # Optional: intercept navigation to enforce allowlist.
                if name == "navigate":
                    url = args.get("url", "")
                    if not _allowed(url):
                        log("blocked", action=name, url=url)
                        continue

                res = run_action(page, name, args)
                log("action", name=name, args=args, result=res)

        # Save a final HTML snapshot for debugging.
        out = Path(os.environ.get("CU_HTML_OUT", ".ci/computer_use_final.html"))
        out.parent.mkdir(parents=True, exist_ok=True)
        out.write_text(page.content(), encoding="utf-8")
        browser.close()
        print(out)
    return 0

if __name__ == "__main__":
    raise SystemExit(main())

import { execSync } from "node:child_process";

const out = [];
function run(cmd, opts={}) {
  try {
    const r = execSync(cmd, { stdio: "pipe", encoding: "utf8", ...opts });

Create router/src/openai-proxy.ts:

import express from "express";
import fetch from "node-fetch";

const app = express();
app.use(express.json({ limit: "5mb" }));

// Choose a backend by ?target=grok|openai|anthropic|alt|local (default: openai)
function pickTarget(q: string | undefined) {
  switch ((q || "openai").toLowerCase()) {
    case "grok":      return { base: process.env.XAI_BASE_URL!, key: process.env.XAI_API_KEY!, model: process.env.XAI_MODEL! };
    case "anthropic": return { base: "https://api.anthropic.com/v1", key: process.env.ANTHROPIC_API_KEY!, model: process.env.ANTHROPIC_MODEL! };
    case "alt":       return { base: process.env.ALT_BASE_URL!, key: process.env.ALT_API_KEY!, model: process.env.ALT_MODEL! };
    case "local":     return { base: process.env.LOCAL_BASE_URL!, key: "no-key", model: process.env.LOCAL_MODEL! };
    default:          return { base: process.env.OPENAI_BASE_URL!, key: process.env.OPENAI_API_KEY!, model: process.env.OPENAI_MODEL! };
  }
}

// OpenAI-compatible chat endpoint for tools that expect /chat/completions
app.post("/v1/chat/completions", async (req, res) => {
  try {
    const target = pickTarget(req.query.target as string);
    const body = { ...req.body };
    if (!body.model) body.model = target.model;

    const headers: Record<string,string> = { "Content-Type": "application/json" };
    if (target.key && target.key !== "no-key") headers["Authorization"] = `Bearer ${target.key}`;
    // Anthropic is not strictly OpenAI-compatible; if you need pure Anthropic, call their native API from your app.
    // For Cline/Continue, keep to OpenAI-compatible providers (OpenAI, xAI, Together, Ollama).

    const r = await fetch(`${target.base}/chat/completions`, { method: "POST", headers, body: JSON.stringify(body) });
    const text = await r.text();
    res.status(r.status).send(text);
  } catch (e:any) {
    res.status(500).json({ error: e?.message || String(e) });
  }
});

const port = Number(process.env.PORT || 8787);
app.listen(port, () => console.log(`[router] OpenAI proxy on :${port}`));

A) Code review with blocking flag (.github/workflows/cursor-code-review.yml)

name: Cursor Code Review
on:
  pull_request:
    types: [opened, synchronize, reopened, ready_for_review]

permissions:
  contents: read
  pull-requests: write

env:
  MODEL: gpt-5
  BLOCKING_REVIEW: 'true'   # flip to 'false' to make non-blocking

jobs:
  review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with: { fetch-depth: 0, ref: ${{ github.event.pull_request.head.sha }} }

      - name: Install Cursor CLI
        run: |
          curl https://cursor.com/install -fsS | bash
          echo "$HOME/.cursor/bin" >> $GITHUB_PATH

      - name: Configure agent permissions
        run: |
          mkdir -p .cursor
          cat .cursor/cli.json || true

      - name: Perform code review
        env:
          CURSOR_API_KEY: ${{ secrets.CURSOR_API_KEY }}
          GH_TOKEN: ${{ github.token }}
          BLOCKING_REVIEW: ${{ env.BLOCKING_REVIEW }}
        run: |
          cursor-agent --force --model "$MODEL" --output-format=text --print "
          You are in a GitHub runner. Use gh for comments (token set).
          Review this PR diff and leave concise inline comments (max 10) for 🚨 Critical / 🔒 Security / ⚠️ Logic / ⚡ Perf.
          After commenting, print the single line:
          RESULT_CRITICAL=<true|false>
          where true means at least one 🚨 or 🔒 was posted.
          " | tee .ci/review.out

          crit=$(grep -oE 'RESULT_CRITICAL=(true|false)' .ci/review.out | tail -n1 | cut -d= -f2)
          if [ -z "$crit" ]; then crit=false; fi
          echo "CRITICAL_ISSUES_FOUND=$crit" >> $GITHUB_ENV

      - name: Check blocking review results
        if: env.BLOCKING_REVIEW == 'true'
        run: |
          echo "CRITICAL_ISSUES_FOUND: ${CRITICAL_ISSUES_FOUND:-unset}"
          if [ "${CRITICAL_ISSUES_FOUND:-false}" = "true" ]; then
            echo "❌ Critical issues found. Failing as blocking."; exit 1;
          else
            echo "✅ No blocking issues."
          fi

name: Conditional Auto-Merge
on:
  pull_request:

permissions:
  contents: write
  pull-requests: write

jobs:
  maybe-merge:
    if: contains(github.event.pull_request.labels.*.name, 'automerge')
    runs-on: ubuntu-latest
    steps:
      - name: Ensure all required checks passed
        env: { GH_TOKEN: ${{ github.token }} }
        run: |
          PR=${{ github.event.pull_request.number }}
          STATE=$(gh pr view $PR --json mergeStateStatus --jq '.mergeStateStatus')
          echo "mergeStateStatus=$STATE"
          test "$STATE" = "CLEAN" || { echo "::error::Checks not clean"; exit 2; }

      - name: Ensure no blocking review flags
        run: |
          # Convention: review job sets a status check context "cursor-code-review"
          echo "Assuming cursor-code-review job enforces BLOCKING_REVIEW."
          # If you store a flag artifact/comment, query it here instead.

      - name: Merge (squash)
        env: { GH_TOKEN: ${{ github.token }} }
        run: |
          gh pr merge ${{ github.event.pull_request.number }} --squash --delete-branch

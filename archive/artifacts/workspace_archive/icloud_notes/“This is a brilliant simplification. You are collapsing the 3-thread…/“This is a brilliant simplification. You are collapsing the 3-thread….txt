# convergence_refiner.py
from dataclasses import dataclass
from typing import Dict, List, Callable
import asyncio
import re

@dataclass
class ConvergenceConfig:
    models_order: List[str] = ("grok", "gemini", "gpt")
    max_rounds: int = 6
    sim_threshold: float = 0.92  # cosine or ROUGE-L proxy
    stable_rounds: int = 2       # consecutive rounds above threshold

def normalize(t: str) -> str:
    t = re.sub(r'\s+', ' ', t).strip().lower()
    return t

def simple_similarity(a: str, b: str) -> float:
    # fast, cheap proxy (token Jaccard). Swap with embeddings later.
    A = set(normalize(a).split())
    B = set(normalize(b).split())
    return len(A & B) / max(1, len(A | B))

class RotatingRefiner:
    def __init__(self, call_grok: Callable, call_gemini: Callable, call_gpt: Callable, call_claude: Callable, cfg: ConvergenceConfig = ConvergenceConfig()):
        self.calls = {"grok": call_grok, "gemini": call_gemini, "gpt": call_gpt, "claude": call_claude}
        self.cfg = cfg

    async def _refine_once(self, prompt: str, model: str) -> str:
        inst = (
            "You are refining a research sub-query. "
            "Return ONLY the refined prompt text. If no improvement, return EXACTLY: NO_CHANGE.\n\n"
            f"CURRENT:\n{prompt}"
        )
        out = await self.calls[model](inst)
        return prompt if out.strip() == "NO_CHANGE" else out.strip()

    async def refine_until_converged(self, seed_from_claude: str) -> Dict:
        history = [seed_from_claude]
        stable = 0

        for round_idx in range(1, self.cfg.max_rounds + 1):
            cur = history[-1]
            nxt = cur
            for m in self.cfg.models_order:
                nxt = await self._refine_once(nxt, m)

            sim = simple_similarity(cur, nxt)
            history.append(nxt)

            if sim >= self.cfg.sim_threshold:
                stable += 1
            else:
                stable = 0

            if stable >= self.cfg.stable_rounds:
                break

        return {
            "final_prompt": history[-1],
            "rounds": len(history)-1,
            "converged": stable >= self.cfg.stable_rounds,
            "history": history
        }

    async def final_publish_check(self, original_query: str, stitched_answer: str) -> Dict:
        gate = (
            "You are the publication gate. Before publishing, you MUST produce:\n"
            "1) REASON_BEFORE_PUBLISH: one paragraph explaining why the answer is ready (sources, consistency, coverage, risks).\n"
            "2) PUBLISH: YES or NO (uppercase). If NO, list blocking issues.\n\n"
            f"ORIGINAL QUERY:\n{original_query}\n\n"
            f"ANSWER DRAFT:\n{stitched_answer}"
        )
        verdict = await self.calls["claude"](gate)
        publish = "PUBLISH: YES" in verdict.upper()
        return {"verdict_text": verdict, "publish": publish}

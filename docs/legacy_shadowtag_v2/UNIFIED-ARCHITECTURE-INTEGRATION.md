# src/deployment/edge_agent_scheduler.py

from src.aerospace.infrastructure import EdgeMeshArchitecture
from src.agents import DebateAgent
from src.pnkln import Cor

class EdgeAgentScheduler:
    """
    Deploy Pinkln agents across ShadowTag-v2 edge mesh.

    Routes debates to nearest cell tower GPU for minimum latency.
    """

    def __init__(self, mesh: EdgeMeshArchitecture):
        self.mesh = mesh
        self.cor = Cor()  # Routing optimizer

    async def deploy_agent_to_tower(
        self,
        agent: DebateAgent,
        tower_id: str
    ):
        """Deploy agent to specific cell tower GPU"""
        tower = self.mesh.get_tower_node(tower_id)

        # Upload agent model to tower GPU
        await self.upload_agent_model(
            agent=agent,
            gpu_endpoint=tower.gpu_pod.endpoint,
            resources=tower.gpu_pod.available_resources
        )

        # Register with COR routing
        self.cor.register_agent(
            agent_id=agent.id,
            location=tower.location,
            capabilities=agent.capabilities,
            glicko_rating=agent.rating.get_rating()
        )

    async def route_debate_to_nearest_gpu(
        self,
        user_location: dict,
        debate_request: dict
    ) -> str:
        """Route debate to nearest cell tower GPU"""

        # Find nearest tower with available GPU
        nearest_tower = self.cor.find_nearest_tower(user_location)

        # Check if agents deployed
        agents = self.cor.get_deployed_agents(nearest_tower.tower_id)

        if not agents:
            # Deploy agents on-demand
            await self.deploy_debate_agents(nearest_tower.tower_id)

        # Execute debate on local GPU
        result = await self.execute_debate_on_edge(
            tower_id=nearest_tower.tower_id,
            debate_request=debate_request
        )

        return result

    def calculate_latency_savings(self):
        """Calculate latency improvement from edge deployment"""

        # Cloud baseline: 150ms
        cloud_latency = 150.0

        # Edge: User → Tower GPU (5-10ms) + Inference (20ms)
        edge_latency = 30.0

        improvement = (cloud_latency - edge_latency) / cloud_latency

        return {
            "cloud_baseline_ms": cloud_latency,
            "edge_mesh_ms": edge_latency,
            "improvement_percent": improvement * 100,
            "cost_savings_percent": 97.0  # From HANDOFF_SUMMARY
        }

%%python
def mcarlo_rev(n,base,sd,yrs=5,gr=0.6): 
    r=[]; 
    for _ in range(n):
        b=max(0,random.gauss(base,sd))
        v=b*((1+gr)**yrs)
        r.append(v)
    return r
def mcarlo_val(n,rev_mult,rev_samples): 
    out=[max(0,rev_mult*rv) for rv in rev_samples]; 
    return {"mean":st.mean(out),"p10":prc(out,0.1),"p50":prc(out,0.5),"p90":prc(out,0.9),"max":max(out)}
def mcarlo_bundle(cfg): 
    vals=[]; comps={}
    for k,v in cfg.items():
        rs=mcarlo_rev(v["n"],v["base"],v["sd"],v.get("yrs",5),v.get("gr",0.6))
        vv=mcarlo_val(v["n"],v["mult"],rs);comps[k]=vv;vals.append(vv["mean"])
    t=sum(vals)
    return {"components":comps,"sum_mean":t}

class Verdict:
    def __init__(s):s.q=[]
    def add(s,t,dl,prio):s.q.append({"t":t,"dl":dl,"p":prio,"k":0,"d":False})
    def tick(s,now): 
        for i in s.q:
            if i["d"]:continue
            if now>=i["dl"]:i["k"]=2
            elif (i["dl"]-now)<=900:i["k"]=1
    def next(s): 
        u=[x for x in s.q if not x["d"]]
        u.sort(key=lambda x:(-x["k"],-x["p"])); 
        return u[0] if u else None
    def done(s,t): 
        for i in s.q:

import numpy as np
def odor_sim(n=128,src=[(64,64,1.0)],k=0.92,fx=0.02): 
    f=np.zeros((n,n),float)
    for _ in range(256):
        nf=np.copy(f)
        nf[1:-1,1:-1]=(f[1:-1,1:-1]*k+fx*(f[:-2,1:-1]+f[2:,1:-1]+f[1:-1,:-2]+f[1:-1,2:])) 
        for (x,y,s) in src:nf[x%n,y%n]+=s
        f=nf
    return f
def odor_score(f,mask=None): 
    if mask is None:return float(f.mean())
    return float((f*mask).mean())

import time
class Verdict:
    def __init__(s):s.q=[]
    def add(s,t,dl,prio):s.q.append({"t":t,"dl":dl,"p":prio,"k":0,"d":False})
    def tick(s,now): 
        for i in s.q:
            if i["d"]:continue
            if now>=i["dl"]:i["k"]=2
            elif (i["dl"]-now)<=900:i["k"]=1
    def next(s): 
        u=[x for x in s.q if not x["d"]]
        u.sort(key=lambda x:(-x["k"],-x["p"])); 
        return u[0] if u else None
    def done(s,t): 
        for i in s.q:
            if i["t"]==t:i["d"]=True;return True
        return False

%%python
import numpy as np
def odor_sim(n=128,src=[(64,64,1.0)],k=0.92,fx=0.02): 
    f=np.zeros((n,n),float)
    for _ in range(256):
        nf=np.copy(f)
        nf[1:-1,1:-1]=(f[1:-1,1:-1]*k+fx*(f[:-2,1:-1]+f[2:,1:-1]+f[1:-1,:-2]+f[1:-1,2:])) 
        for (x,y,s) in src:nf[x%n,y%n]+=s
        f=nf
    return f
def odor_score(f,mask=None): 
    if mask is None:return float(f.mean())
    return float((f*mask).mean())

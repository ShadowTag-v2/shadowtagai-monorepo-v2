/ops/collect-thread.mjs

import { promises as fs } from "node:fs";
import { glob } from "node:fs/promises";
const out = { items: [] };
for await (const p of glob("docs/thread/**/*.md")) {
  const text = await fs.readFile(p, "utf8");
  out.items.push({ path: p, text });
}
await fs.mkdir("build", { recursive: true });
await fs.writeFile("build/thread.json", JSON.stringify(out, null, 2));
console.log(`Collected ${out.items.length} thread files → build/thread.json`);

/ops/mega-rollup.mjs

import { promises as fs } from "node:fs";
import path from "node:path";
import { glob } from "node:fs/promises";

const cfg = JSON.parse(await fs.readFile("./ops/rollup.config.json", "utf8"));
const chunks = [];

async function pull(files) {
  for await (const f of glob(files)) {
    const t = await fs.readFile(f, "utf8");
    chunks.push(`\n---\n\n## Source: ${f}\n\n${t.trim()}\n`);
  }
}

for (const pattern of cfg.sources) await pull(pattern);

const header = `# Mega Roll-Up\n\n> Generated: ${new Date().toISOString()}\n\n`;
const body = chunks.join("\n");
await fs.writeFile(cfg.output, header + body);
console.log(`Wrote ${cfg.output} (${chunks.length} sections)`);

4) Release with artifacts (attach .node binaries)

.github/workflows/release-with-artifacts.yml

name: release-with-artifacts
on:
  push:
    tags: ["v*.*.*", "blake3-native-v*.*.*"]
jobs:
  build:
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        include:
          - os: ubuntu-latest
            target: linux-x64
          - os: windows-latest
            target: win-x64
          - os: macos-13
            target: macos-x64
          - os: macos-14
            target: macos-arm64
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '22' }
      - run: npm ci
      - name: Build native
        run: |
          npm run -w router/packages/blake3-native build -- --target=${{ matrix.target }}
      - name: Pack artifact
        run: |
          mkdir -p dist/${{ matrix.target }}
          cp -r router/packages/blake3-native/build/*.node dist/${{ matrix.target }}/
      - uses: softprops/action-gh-release@v2
        with:
          files: dist/${{ matrix.target }}/*.node
        env:
          GITHUB_[CREDENTIAL_REDACTED] secrets.GITHUB_TOKEN }}
  publish-npm:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '22', registry-url: 'https://registry.npmjs.org' }
      - run: npm ci
      - run: npm publish -w router/packages/blake3-native --access public
        env: { NODE_AUTH_[CREDENTIAL_REDACTED] secrets.NPM_TOKEN }} }

5) Ultra release helper

/scripts/ultra-release.mjs

import { execSync } from "node:child_process";
import { readFileSync, writeFileSync } from "node:fs";
const pkgPath = "router/packages/blake3-native/package.json";
const pkg = JSON.parse(readFileSync(pkgPath, "utf8"));
const [maj, min, pat] = pkg.version.split(".").map(Number);
pkg.version = [maj, min, pat + 1].join(".");
writeFileSync(pkgPath, JSON.stringify(pkg, null, 2));
execSync("git add -A && git commit -m \"chore(release): v" + pkg.version + "\"", { stdio: "inherit" });
const tag = (pkg.name.includes("blake3-native") ? "blake3-native-v" : "v") + pkg.version;
execSync(`git tag ${tag} && git push && git push --tags`, { stdio: "inherit" });
console.log(`Tagged ${tag}. GitHub Actions will build & attach .node binaries.`);
